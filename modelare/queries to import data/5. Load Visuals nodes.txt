//5.Load Visuals nodes
LOAD CSV WITH HEADERS
FROM 'file:///visuals.csv'
AS line
MATCH (g:Game {id: line.id})
MERGE (g)-[:HAS]->(v:Visuals)
    ON CREATE SET
          v.screenshots = split(substring(line.screenshots, 1, size(line.screenshots) - 2), ', '),
          v.summary = line.summary,
          v.storyline = line.storyline,
          v.cover = line.cover
